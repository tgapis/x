import asyncio
import random
from ferogram import Client, raw

app = Client("my_session", api_id=12345, api_hash="0123456789abcdef0123456789abcdef")

async def main():
    await app.start()
    result = await app(raw.functions.messages.SendEncryptedFile(
        silent=True,
        peer=raw.types.InputEncryptedChat(chat_id=0, access_hash=0),
        random_id=random.randrange(-2**63, 2**63),
        data=b'arbitrary\x7f data',
        file=raw.types.InputEncryptedFileEmpty()
    ))
    print(result)

asyncio.run(main())
