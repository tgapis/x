Example

▶ Minimal (required fields only) 
from ferogram import raw, types

result = await client(raw.functions.messages.Search(
    peer='username',
    q='some string here',
    filter=types.MessagesFilter(),
    min_date=42,
    max_date=42,
    offset_id=42,
    add_offset=0,
    limit=100,
    max_id=0,
    min_id=0,
    hash=0
))
print(result)

from ferogram import raw, types

result = await client(raw.functions.messages.Search(
    peer='username',
    q='some string here',
    from_id='username',
    saved_peer_id='username',
    saved_reaction=types.Vector<Reaction>(),
    top_msg_id=42,
    filter=types.MessagesFilter(),
    min_date=42,
    max_date=42,
    offset_id=42,
    add_offset=0,
    limit=100,
    max_id=0,
    min_id=0,
    hash=0
))
print(result)
