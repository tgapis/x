▶ Full API 
import asyncio
import random
from ferogram import Client, raw

app = Client("my_session", api_id=12345, api_hash="0123456789abcdef0123456789abcdef")

async def main():
    await app.start()
    result = await app(raw.functions.messages.SendPaidReaction(
        peer=raw.types.InputPeerSelf(),
        msg_id=42,
        count=42,
        random_id=random.randrange(-2**63, 2**63),
        private=raw.types.PaidReactionPrivacyDefault()
    ))
    print(result)

asyncio.run(main())
