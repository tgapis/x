import asyncio
from ferogram import Client, raw

app = Client("my_session", api_id=12345, api_hash="0123456789abcdef0123456789abcdef")

async def main():
    await app.start()
    result = await app(raw.functions.messages.RequestUrlAuth(
        peer=raw.types.InputPeerSelf(),
        msg_id=42,
        button_id=42,
        url='some string here',
        in_app_origin='some string here'
    ))
    print(result)

asyncio.run(main())
