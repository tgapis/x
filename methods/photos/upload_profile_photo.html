Example

▶ Minimal (required fields only) 
from ferogram import raw, types

result = await client(raw.functions.photos.UploadProfilePhoto())
print(result)
▶ Full (all fields) 
from ferogram import raw, types

result = await client(raw.functions.photos.UploadProfilePhoto(
    fallback=True,
    bot='username',
    file=types.InputFile(),
    video=types.InputFile(),
    video_start_ts=7.13,
    video_emoji_markup=types.VideoSize()
))
print(result)
