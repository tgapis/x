Example

▶ Minimal (required fields only) 
from ferogram import raw, types

result = await client(raw.functions.account.CreateTheme(
    slug='some string here',
    title='My title'
))

▶ Full (all fields) 
from ferogram import raw, types

result = await client(raw.functions.account.CreateTheme(
    slug='some string here',
    title='My title',
    document=types.InputDocument(),
    settings=types.Vector<InputThemeSettings>()
))
print(result)
